delay:
        PUSH {R1-R5, LR}     ; save registers we use

        LDR R1, .Time        ; start time

delay_loop:
        LDR R2, .Time        ; current time
        SUB R3, R2, R1       ; elapsed = current - start
        CMP R3, R0
        BLT delay_loop       ; keep waiting

        POP {R1-R5, LR}
        BX LR                ; return
